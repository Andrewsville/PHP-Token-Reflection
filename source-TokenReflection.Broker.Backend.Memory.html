 89:      */
 90:     public function getClass($className)
 91:     {
 92:         static $declared = array();
 93:         if (empty($declared)) {
 94:             $declared = array_flip(array_merge(get_declared_classes(), get_declared_interfaces()));
 95:         }
 96: 
 97:         $className = ltrim($className, '\\');
 98:         try {
 99:             $ns = $this->getNamespace(
100:                 ($boundary = strrpos($className, '\\'))
101:                     ? substr($className, 0, $boundary)        // Class within a namespace
102:                     : TokenReflection\ReflectionNamespace::NO_NAMESPACE_NAME  // Class without a namespace
103:             );
104: 
105:             return $ns->getClass($className);
106:         } catch (Exception $e) {
107:             if (isset($declared[$className])) {
108:                 $reflection = new Php\ReflectionClass($className, $this->broker);
109:                 if ($reflection->isInternal()) {
110:                     return $reflection;

112:             }

282:     /**
283:      * Prepares and returns used class lists.
284:      *
285:      * @return array
286:      */
287:     protected function parseClassLists()
288:     {
289:         // Initialize the all-classes-cache
290:         $allClasses = array(
291:             self::TOKENIZED_CLASSES => array(),
292:             self::INTERNAL_CLASSES => array(),
293:             self::NONEXISTENT_CLASSES => array()
294:         );
295: 
296:         foreach ($this->namespaces as $namespace) {
297:             foreach ($namespace->getClasses() as $class) {
298:                 $allClasses[self::TOKENIZED_CLASSES][$class->getName()] = $class;
299:             }
300:         }
301: 
302:         foreach ($allClasses[self::TOKENIZED_CLASSES] as $className => $class) {
303:             foreach (array_merge($class->getParentClasses(), $class->getInterfaces()) as $parent) {
304:                 if ($parent->isTokenized()) {
305:                     if (!isset($allClasses[self::TOKENIZED_CLASSES][$parent->getName()])) {
306:                         throw new RuntimeException(sprintf('Class %s should be tokenized', $parent->getName()));
307:                     }
308:                 } elseif ($parent->isInternal()) {
309:                     $allClasses[self::INTERNAL_CLASSES][$parent->getName()] = $parent;
310:                 } else {
311:                     $allClasses[self::NONEXISTENT_CLASSES][$parent->getName()] = $parent;
312:                 }
313:             }
314:         }
315: 
316:         return $allClasses;
317:     }

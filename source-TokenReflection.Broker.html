 90:     /**
 91:      * Parses a file a returns the appropriate reflection object.
 92:      *
 93:      * @param string $filename Filename
 94:      * @return \TokenReflection\ReflectionFile
 95:      * @throws \TokenReflection\Exception If the requested file does not exist
 96:      */
 97:     public function processFile($filename)
 98:     {
 99:         $realName = realpath($filename);
100:         if (false === $realName) {
101:             throw new Exception(sprintf('File %s does not exist', $filename), Exception::FILE_DOES_NOT_EXIST);
102:         }
103: 
104:         if ($this->backend->isFileProcessed($realName)) {
105:             $tokens = $this->backend->getFileTokens($realName);
106:         } else {
107:             $contents = @file_get_contents($filename);
108:             if (false === $contents) {
109:                 throw new Exception('File is not readable', Exception::FILE_NOT_READABLE);
110:             }
111: 
112:             $contents = str_replace(array("\r\n", "\r"), "\n", $contents);
113: 
114:             static $checkLines = array(T_COMMENT, T_WHITESPACE, T_DOC_COMMENT, T_INLINE_HTML, T_ENCAPSED_AND_WHITESPACE, T_CONSTANT_ENCAPSED_STRING);
115:             $actualLine = 0;
116:             $tmp = null;
117: 
118:             $tokens = @token_get_all($contents);
119:             foreach ($tokens as $index => $token) {
120:                 if (!is_array($token)) {
121:                     if (0 === $index) {
122:                         throw new RuntimeException('The first token has to be a complete one.');
123:                     }
124: 
125:                     $tokens[$index] = array($token, $token, $actualLine);
126:                 } else {
127:                     $actualLine = $token[2];
128:                     if (in_array($token[0], $checkLines)) {
129:                         $actualLine += substr_count($token[1], "\n", $tmp);
130:                     }
131:                 }
132:             }
133:         }
134: 
135:         $reflectionFile = new ReflectionFile($realName, $tokens, $this);
136:         if (!$this->backend->isFileProcessed($realName)) {
137:             $this->backend->addFile($reflectionFile);
138: 
139:             // Clear the cache - leave only tokenized reflections
140:             foreach ($this->cache as $type => $cached) {
141:                 if (!empty($cached)) {
142:                     $this->cache[$type] = array_filter($cached, function(IReflection $reflection) {
143:                         return $reflection->isTokenized();
144:                     });
145:                 }
146:             }
147:         }
148:         return $reflectionFile;
149:     }

152:      * Processes recursively a directory and returns an array of file reflection objects.
153:      *
154:      * @param string $path Directora path
155:      * @return array
156:      * @throws \TokenReflection\Exception If the requested directory does not exist
157:      */
158:     public function processDirectory($path)
159:     {
160:         $realPath = realpath($path);
161:         if (false === $realPath) {
162:             throw new Exception(sprintf('Directory %s does not exist', $path), Exception::FILE_DOES_NOT_EXIST);
163:         }
164: 
165:         $result = array();
166:         foreach (new RecursiveIteratorIterator(new RecursiveDirectoryIterator($realPath)) as $entry) {
167:             if ($entry->isFile()) {
168:                 $result[$entry->getPathName()] = $this->processFile($entry->getPathName());
169:             }
170:         }
171: 
172:         return $result;

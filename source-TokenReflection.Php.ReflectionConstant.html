359:      * @param string $constant Constant name
360:      * @param boolean $return Return the export instead of outputting it
361:      * @return string|null
362:      * @throws \TokenReflection\Exception\Runtime If requested parameter doesn't exist
363:      */
364:     public static function export(Broker $broker, $class, $constant, $return = false)
365:     {
366:         $className = is_object($class) ? get_class($class) : $class;
367:         $constantName = $constant;
368: 
369:         if (null === $className) {
370:             $constant = $broker->getConstant($constantName);
371:             if (null === $constant) {
372:                 throw new Exception\Runtime(sprintf('Constant %s does not exist.', $constantName), Exception\Runtime::DOES_NOT_EXIST);
373:             }
374:         } else {
375:             $class = $broker->getClass($className);
376:             if ($class instanceof Dummy\ReflectionClass) {
377:                 throw new Exception\Runtime(sprintf('Class %s does not exist.', $className), Exception\Runtime::DOES_NOT_EXIST);
378:             }
379:             $constant = $class->getConstantReflection($constantName);
380:         }
381: 
382:         if ($return) {
383:             return $constant->__toString();
384:         }
385: 
386:         echo $constant->__toString();
387:     }

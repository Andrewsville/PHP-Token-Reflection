 37:     /**
 38:      * Parses reflection object documentation.
 39:      *
 40:      * @param ReflectionBase $reflection Reflection object
 41:      * @return array
 42:      */
 43:     public static function parse(ReflectionBase $reflection)
 44:     {
 45:         $docblock = $reflection->getInheritedDocComment();
 46: 
 47:         if (false === $docblock) {
 48:             return array();
 49:         }
 50: 
 51:         // Preprocess docblock
 52:         $docblock = trim(preg_replace(array('~^/\s*\*\*~', '~\*/$~'), '', $docblock));
 53:         $docblock = array_map(function($line) {
 54:             return preg_replace('~^\\s*\\*\\s*~', '', trim($line), 1);
 55:         }, explode("\n", $docblock));
 56: 
 57:         // Parse docblock
 58:         $result = array();
 59:         $name = self::SHORT_DESCRIPTION;
 60:         foreach ($docblock as $line) {
 61:             if (preg_match('~^@\\s*([\\S]+)\\s*(.*)~', $line, $matches)) {
 62:                 if (!isset($result['PARAMS'])) {
 63:                     $result['PARAMS'] = array();
 64:                 }
 65: 
 66:                 $name = strtolower($matches[1]);
 67: 
 68:                 if (!isset($result['PARAMS'][$name])) {
 69:                     $result['PARAMS'][$name] = array();
 70:                 }
 71:                 $result['PARAMS'][$name][] = $matches[2];
 72:             } else {
 73:                 if (empty($line)) {
 74:                     if (self::SHORT_DESCRIPTION === $name) {
 75:                         // End of main description
 76:                         $name = self::LONG_DESCRIPTION;
 77:                         continue;
 78:                     } else {
 79:                         $line = "\n";
 80:                     }
 81:                 }
 82: 
 83:                 if (self::SHORT_DESCRIPTION === $name || self::LONG_DESCRIPTION === $name) {
 84:                     if (!isset($result[$name])) {
 85:                         $result[$name] = $line;
 86:                     } else {
 87:                         $result[$name] .= "\n" . $line;
 88:                     }
 89:                 } else {
 90:                     if (is_array($result['PARAMS'][$name])) {
 91:                         $index = count($result['PARAMS'][$name]) - 1;
 92:                         $result['PARAMS'][$name][$index] .= ' ' . trim($line);
 93:                     } else {
 94:                         $result['PARAMS'][$name] .= ' ' . trim($line);
 95:                     }
 96:                 }
 97:             }
 98:         }
 99: 
100:         array_walk_recursive($result, function(&$value) {
101:             $value = trim($value);
102:         });

105:     }

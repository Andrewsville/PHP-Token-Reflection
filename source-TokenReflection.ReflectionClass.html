 374:     /**
 375:      * Returns if the class implements the given interface.
 376:      *
 377:      * @param string|object $interface Interface name or reflection object
 378:      * @return boolean
 379:      * @throws \TokenReflection\Exception\Runtime If the provided parameter is not an interface
 380:      */

 382:     {
 383:         if (is_object($interface)) {
 384:             if (!$interface instanceof InternalReflectionClass && !$interface instanceof IReflectionClass) {
 385:                 throw new Exception\Runtime(sprintf('Parameter must be a string or an instance of class reflection, "%s" provided.', get_class($interface)), Exception\Runtime::INVALID_ARGUMENT);
 386:             }

 390:             if (!$interface->isInterface()) {

 392:             }

 396:                 throw new Exception\Runtime(sprintf('"%s" is not an interface.', $interface), Exception\Runtime::INVALID_ARGUMENT);
 397:             }

 428:     public function getInterfaceNames()
 429:     {
 430:         $parentClass = $this->getParentClass();

 432:         $names = false !== $parentClass ? array_reverse(array_flip($parentClass->getInterfaceNames())) : array();

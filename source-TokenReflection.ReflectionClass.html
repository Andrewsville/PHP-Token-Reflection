 139:     /**
 140:      * Parses child reflection objects from the token stream.
 141:      *
 142:      * @param \TokenReflection\Stream $tokenStream Token substream
 143:      * @return \TokenReflection\ReflectionBase
 144:      */
 145:     protected function parseChildren(Stream $tokenStream, IReflection $parent)
 146:     {
 147:         while (true) {
 148:             switch ($tokenStream->getType()) {
 149:                 case '}':
 150:                     $tokenStream->next();
 151:                     break 2;
 152:                 case null:
 153:                     break 2;
 154:                 case T_PUBLIC:
 155:                 case T_PRIVATE:
 156:                 case T_PROTECTED:
 157:                 case T_STATIC:
 158:                 case T_VARIABLE:
 159:                     static $searching = array(T_VARIABLE, T_FUNCTION);
 160: 
 161:                     $position = $tokenStream->key();
 162:                     while (null !== ($type = $tokenStream->getType($position++)) && !in_array($type, $searching)) {
 163:                         $position++;
 164:                     }
 165: 
 166:                     if (T_VARIABLE === $type) {
 167:                         $property = new ReflectionProperty($tokenStream, $this->getBroker(), $this);
 168:                         $this->properties[$property->getName()] = $property;
 169:                         break;
 170:                     }
 171:                     // Break missing on purpose
 172:                 case T_FINAL:
 173:                 case T_ABSTRACT:
 174:                 case T_FUNCTION:
 175:                     $method = new ReflectionMethod($tokenStream, $this->getBroker(), $this);
 176:                     $this->methods[$method->getName()] = $method;
 177:                     break;
 178:                 case T_CONST:
 179:                     $tokenStream->skipWhitespaces();
 180:                     while ($tokenStream->is(T_STRING)) {
 181:                         $constant = new ReflectionConstant($tokenStream, $this->getBroker(), $this);
 182:                         $this->constants[$constant->getName()] = $constant;
 183:                         $tokenStream->skipWhitespaces();
 184:                     }
 185:                     break;
 186:                 case T_VAR:
 187:                 case T_VARIABLE:
 188:                     $property = new ReflectionProperty($tokenStream, $this->getBroker(), $this);
 189:                     $this->properties[$property->getName()] = $property;
 190:                     break;
 191:                 default:
 192:                     $tokenStream->skipWhitespaces();
 193:             }
 194:         }
 195: 
 196:         return $this;
 197:     }

 720:      * Returns if the class implements the given interface.
 721:      *
 722:      * @param string|object $interface Interface name or reflection object
 723:      * @return boolean
 724:      */
 725:     public function implementsInterface($interface) {
 726:         if (is_object($interface)) {
 727:             if ($interface instanceof InternalReflectionClass || $interface instanceof IReflectionClass) {
 728:                 $interfaceName = $interface->getName();
 729:             } else {
 730:                 throw new InvalidArgumentException('Parameter must be a string or an instance of class reflection');

 735:             }
 736:         } else {

 738:             if (!$reflection->isInterface()) {
 739:                 throw new RuntimeException(sprintf('%s is not an interface.', $interface));
 740:             }

 743:         }
 744: 
 745:         return in_array($interfaceName, $this->getInterfaceNames());

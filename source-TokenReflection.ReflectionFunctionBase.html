367:                                 while (T_VARIABLE === $type) {

371:                                     $type = $tokenStream->skipWhitespaces()->getType();

378:                                                 case '[':

384:                                                 case ')':
385:                                                 case ']':
386:                                                 case '}':

388:                                                     break;
389:                                                 case ';':
390:                                                 case ',':

394:                                             }

397:                                             $type = $tokenStream->skipWhitespaces()->getType();
398:                                         }
399: 
400:                                         if (!$tokenStream->valid()) {

402:                                         }
403:                                     }

405:                                     $this->staticVariablesDefinition[substr($variableName, 1)] = $variableDefinition;
406: 
407:                                     if (',' === $type) {
408:                                         $type = $tokenStream->skipWhitespaces()->getType();
409:                                     } else {
410:                                         break;
411:                                     }
412:                                 }
413: 
414:                                 break;
415:                             case T_FUNCTION:
416:                                 // Anonymous function -> skip to its end
417:                                 if (!$tokenStream->find('{')) {
418:                                     throw new Exception\Parse('Could not find beginning of the anonymous function.', Exception\Parse::PARSE_CHILDREN_ERROR);

420:                                 // Break missing intentionally
421:                             case '{':

437:             }
438: 
439:             return $this;

370:     /**
371:      * Parses the propety default value.
372:      *
373:      * @param \TokenReflection\Stream $tokenStream Token substream
374:      * @return \TokenReflection\ReflectionProperty
375:      */
376:     private function parseDefaultValue(Stream $tokenStream)
377:     {
378:         $type = $tokenStream->getType();
379: 
380:         if (';' === $type || ',' === $type) {
381:             // No default value
382:             return $this;
383:         }
384: 
385:         if ('=' === $type) {
386:             $tokenStream->skipWhitespaces();
387:         }
388: 
389:         $level = 0;
390:         while (null !== ($type = $tokenStream->getType())) {
391:             switch ($type) {
392:                 case ',':
393:                     if (0 !== $level) {
394:                         break;
395:                     }
396:                 case ';':
397:                     break 2;
398:                 case ')':
399:                 case ']':
400:                 case '}':
401:                     $level--;
402:                     break;
403:                 case '(':
404:                 case '{':
405:                 case '[':
406:                     $level++;
407: 
408:             }
409: 
410:             $this->defaultValueDefinition .= $tokenStream->getTokenValue();
411:             $tokenStream->next();
412: 
413:         }
414: 
415:         if (',' === $type) {
416:             $tokenStream->next();
417:         } elseif (';' !== $type) {
418:             throw new RuntimeException('Property definition is not terminated properly');
419:         }
420: 
421:         if (self::$parseValueDefinitions) {
422:             // Následuje husťárna (a fucking awesomness follows)
423:             $this->defaultValue = @eval('return ' . $this->defaultValueDefinition . ';');
424:         }
425: 
426:         return $this;
427:     }

429:     /**
430:      * Parses class modifiers (abstract, final) and class type (class, interface).
431:      *
432:      * @param \TokenReflection\Stream $tokenStream Token substream
433:      * @param \TokenReflection\ReflectionClass $class Defining class
434:      * @return \TokenReflection\ReflectionClass
435:      */
436:     private function parseModifiers(Stream $tokenStream, ReflectionClass $class)
437:     {
438:         while (true) {
439:             switch ($tokenStream->getType()) {
440:                 case T_PUBLIC:
441:                 case T_VAR:
442:                     $this->modifiers |= InternalReflectionProperty::IS_PUBLIC;
443:                     break;
444:                 case T_PROTECTED:
445:                     $this->modifiers |= InternalReflectionProperty::IS_PROTECTED;
446:                     break;
447:                 case T_PRIVATE:
448:                     $this->modifiers |= InternalReflectionProperty::IS_PRIVATE;
449:                     break;
450:                 case T_STATIC:
451:                     $this->modifiers |= InternalReflectionProperty::IS_STATIC;
452:                     break;
453:                 default:
454:                     break 2;
455:             }
456: 
457:             $tokenStream->skipWhitespaces();
458:         }
459: 
460:         if (InternalReflectionProperty::IS_STATIC === $this->modifiers) {
461:             $this->modifiers |= InternalReflectionProperty::IS_PUBLIC;
462:         } elseif (0 === $this->modifiers) {
463:             $parentProperties = $class->getOwnProperties();
464:             if (empty($parentProperties)) {
465:                 throw new RuntimeException('No access level defined and no previous defining class property present.');
466:             }
467: 
468:             $sibling = array_pop($parentProperties);
469:             if ($sibling->isPublic()) {
470:                 $this->modifiers = InternalReflectionProperty::IS_PUBLIC;
471:             } elseif ($sibling->isPrivate()) {
472:                 $this->modifiers = InternalReflectionProperty::IS_PRIVATE;
473:             } elseif ($sibling->isProtected()) {
474:                 $this->modifiers = InternalReflectionProperty::IS_PROTECTED;
475:             } else {
476:                 throw new RuntimeException(sprintf('Property sibling %s has no access level defined.', $sibling->getName()));
477:             }
478: 
479:             if ($sibling->isStatic()) {
480:                 $this->modifiers |= InternalReflectionProperty::IS_STATIC;
481:             }
482:         }
483: 
484:         return $this;
485:     }

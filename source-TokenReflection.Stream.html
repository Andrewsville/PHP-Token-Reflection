176:      * Returns the position of the token with the matching bracket.
177:      *
178:      * @return \TokenReflection\Stream
179:      * @throws \TokenReflection\Exception\Runtime If out of the array

183:     public function findMatchingBracket()
184:     {

187:             '{' => '}',

190:             T_DOLLAR_OPEN_CURLY_BRACES => '}'

199:         $bracket = $this->tokens[$this->position][0];
200: 
201:         if (!isset($brackets[$bracket])) {
202:             throw new Exception\Runtime(sprintf('There is no usable bracket at position "%d" in file "%s".', $position, $this->fileName), Exception\Runtime::DOES_NOT_EXIST);
203:         }
204: 
205:         $searching = $brackets[$bracket];

207:         $level = 0;

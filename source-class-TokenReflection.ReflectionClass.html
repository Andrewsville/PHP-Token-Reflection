 697:      * @return array
 698:      * @throws \TokenReflection\Exception\RuntimeException If trait method was already imported.
 699:      */
 700:     public function getTraitMethods($filter = null)
 701:     {
 702:         $methods = array();
 703: 
 704:         foreach ($this->getOwnTraits() as $trait) {
 705:             $traitName = $trait->getName();
 706:             foreach ($trait->getMethods(null) as $traitMethod) {
 707:                 $methodName = $traitMethod->getName();
 708: 
 709:                 $imports = array();
 710:                 if (isset($this->traitImports[$traitName . '::' . $methodName])) {
 711:                     $imports = $this->traitImports[$traitName . '::' . $methodName];
 712:                 }
 713:                 if (isset($this->traitImports[$methodName])) {
 714:                     $imports = empty($imports) ? $this->traitImports[$methodName] : array_merge($imports, $this->traitImports[$methodName]);
 715:                 }
 716: 
 717:                 foreach ($imports as $import) {
 718:                     if (null !== $import) {
 719:                         list($newName, $accessLevel) = $import;
 720: 
 721:                         if ('' === $newName) {
 722:                             $newName = $methodName;
 723:                             $imports[] = null;
 724:                         }
 725: 
 726:                         if (!isset($this->methods[$newName])) {
 727:                             if (isset($methods[$newName])) {
 728:                                 throw new Exception\RuntimeException(sprintf('Trait method "%s" was already imported.', $newName), Exception\RuntimeException::ALREADY_EXISTS, $this);
 729:                             }
 730: 
 731:                             $methods[$newName] = $traitMethod->alias($this, $newName, $accessLevel);
 732:                         }
 733:                     }
 734:                 }
 735: 
 736:                 if (!in_array(null, $imports)) {
 737:                     if (!isset($this->methods[$methodName])) {
 738:                         if (isset($methods[$methodName])) {
 739:                             throw new Exception\RuntimeException(sprintf('Trait method "%s" was already imported.', $methodName), Exception\RuntimeException::ALREADY_EXISTS, $this);
 740:                         }
 741: 
 742:                         $methods[$methodName] = $traitMethod->alias($this);

258:     protected function parseDocComment(Stream $tokenStream, IReflection $parent)
259:     {
260:         if ($this instanceof ReflectionParameter) {
261:             $this->docComment = new ReflectionAnnotation($this);
262:             return $this;
263:         }
264: 
265:         $position = $tokenStream->key();
266: 
267:         if ($tokenStream->is(T_DOC_COMMENT, $position - 1)) {
268:             $value = $tokenStream->getTokenValue($position - 1);
269:             if (self::DOCBLOCK_TEMPLATE_END !== $value) {
270:                 $this->docComment = new ReflectionAnnotation($this, $value);
271:                 $this->startPosition--;
272:             }
273:         } elseif ($tokenStream->is(T_DOC_COMMENT, $position - 2)) {
274:             $value = $tokenStream->getTokenValue($position - 2);
275:             if (self::DOCBLOCK_TEMPLATE_END !== $value) {
276:                 $this->docComment = new ReflectionAnnotation($this, $value);
277:                 $this->startPosition -= 2;
278:             }
279:         } elseif ($tokenStream->is(T_COMMENT, $position - 1) && preg_match('~^' . preg_quote(self::DOCBLOCK_TEMPLATE_START, '~') . '~', $tokenStream->getTokenValue($position - 1))) {

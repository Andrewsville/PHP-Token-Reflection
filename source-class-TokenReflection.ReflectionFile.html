105:                     case T_COMMENT:
106:                         break;

112:                             ->skipWhitespaces()
113:                             ->skipWhitespaces();
114:                         break;

116:                         $docCommentPosition = $docCommentPosition ?: -1;
117:                         break 2;
118:                     default:
119:                         $docCommentPosition = $docCommentPosition ?: -1;
120:                         $this->namespaces[] = new ReflectionFileNamespace($tokenStream, $this->broker, $this);
121:                         break 2;
122:                 }
123: 
124:                 $tokenStream->skipWhitespaces();
125:             }
126: 
127:             while (null !== ($type = $tokenStream->getType())) {
128:                 if (T_NAMESPACE === $type) {
129:                     $this->namespaces[] = new ReflectionFileNamespace($tokenStream, $this->broker, $this);
130:                 } else {
131:                     $tokenStream->skipWhitespaces();
132:                 }
133:             }
134:         }
135:

138:         }

165:      * Returns method modifiers.
166:      *
167:      * @return integer
168:      */
169:     public function getModifiers()
170:     {
171:         if (!$this->modifiersComplete && !($this->modifiers & (self::ACCESS_LEVEL_CHANGED | self::IS_IMPLEMENTED_ABSTRACT))) {
172:             $declaringClass = $this->getDeclaringClass();
173:             $parentClass = $declaringClass->getParentClass();
174:             if (false !== $parentClass && $parentClass->hasMethod($this->name)) {
175:                 $parentClassMethod = $parentClass->getMethod($this->name);
176: 
177:                 // Access level changed
178:                 if (($this->isPublic() || $this->isProtected()) && $parentClassMethod->is(self::ACCESS_LEVEL_CHANGED | InternalReflectionMethod::IS_PRIVATE)) {
179:                     $this->modifiers |= self::ACCESS_LEVEL_CHANGED;
180:                 }
181: 
182:                 // Implemented abstract
183:                 if ($parentClassMethod->isAbstract() && !$this->isAbstract()) {
184:                     $this->modifiers |= self::IS_IMPLEMENTED_ABSTRACT;
185:                 }
186:             } else {
187:                 // Check if it is an implementation of an interface method
188:                 foreach ($declaringClass->getInterfaces() as $interface) {
189:                     if ($interface->hasOwnMethod($this->name)) {
190:                         $this->modifiers |= self::IS_IMPLEMENTED_ABSTRACT;
191:                         break;
192:                     }
193:                 }
194:             }
195: 
196:             // Set if modifiers definition is complete
197:             $this->modifiersComplete = $this->isComplete() || (($this->modifiers & self::IS_IMPLEMENTED_ABSTRACT) && ($this->modifiers & self::ACCESS_LEVEL_CHANGED));

201:     }

314:      * @return \TokenReflection\ReflectionMethod
315:      * @throws \TokenReflection\Exception\Runtime If the method has no prototype.
316:      */
317:     public function getPrototype()
318:     {

473:     /**
474:      * Calls the method on an given object.
475:      *
476:      * @param object $object Class instance
477:      * @param array $args Method parameter values
478:      * @return mixed
479:      * @throws \TokenReflection\Exception\Runtime If it is not possible to invoke the method.
480:      */
481:     public function invokeArgs($object, array $args = array())
482:     {
483:         try {
484:             $declaringClass = $this->getDeclaringClass();
485:             if (!$declaringClass->isInstance($object)) {
486:                 throw new Exception\Runtime(sprintf('Invalid class, "%s" expected "%s" given.', $declaringClass->getName(), get_class($object)), Exception\Runtime::INVALID_ARGUMENT);
487:             }
488: 
489:             if ($this->isPublic()) {
490:                 return call_user_func_array(array($object, $this->getName()), $args);
491:             } elseif ($this->isAccessible()) {
492:                 $refClass = new InternalReflectionClass($object);
493:                 $refMethod = $refClass->getMethod($this->name);
494: 
495:                 $refMethod->setAccessible(true);
496:                 $value = $refMethod->invokeArgs($object, $args);
497:                 $refMethod->setAccessible(false);
498: 
499:                 return $value;

501: 
502:             throw new Exception\Runtime('Only public methods can be invoked.', Exception\Runtime::NOT_ACCESSBILE);
503:         } catch (Exception\Runtime $e) {

571:     /**
572:      * Creates a method alias of the given name and access level for the given class.
573:      *
574:      * @param \TokenReflection\ReflectionClass $parent New parent class
575:      * @param string $name New method name
576:      * @param integer $accessLevel New access level
577:      * @return \TokenReflection\ReflectionMethod
578:      * @throws \TokenReflection\Exception\Parse If an invalid method access level was found.
579:      */
580:     public function alias(ReflectionClass $parent, $name = null, $accessLevel = null)
581:     {
582:         static $possibleLevels = array(InternalReflectionMethod::IS_PUBLIC => true, InternalReflectionMethod::IS_PROTECTED => true, InternalReflectionMethod::IS_PRIVATE => true);
583: 
584:         $method = clone $this;
585: 
586:         $method->declaringClassName = $parent->getName();
587:         if (null !== $name) {
588:             $method->originalName = $this->name;
589:             $method->name = $name;
590:         }
591:         if (null !== $accessLevel) {
592:             if (!isset($possibleLevels[$accessLevel])) {
593:                 throw new Exception\Parse(sprintf('Invalid method access level: "%s".', $accessLevel), Exception\Parse::PARSE_CHILDREN_ERROR);
594:             }
595: 
596:             $method->modifiers &= ~(InternalReflectionMethod::IS_PUBLIC | InternalReflectionMethod::IS_PROTECTED | InternalReflectionMethod::IS_PRIVATE);
597:             $method->modifiers |= $accessLevel;
598: 
599:             $method->originalModifiers = $this->getModifiers();
600:         }
601: 
602:         foreach ($this->parameters as $parameterName => $parameter) {
603:             $method->parameters[$parameterName] = $parameter->alias($method);
604:         }
605: 
606:         return $method;
607:     }

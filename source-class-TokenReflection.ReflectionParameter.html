501:      * Parses the type hint.
502:      *
503:      * @param \TokenReflection\Stream\StreamBase $tokenStream Token substream
504:      * @return \TokenReflection\ReflectionParameter
505:      * @throws \TokenReflection\Exception\ParseException If the type hint class name could not be determined.
506:      */
507:     private function parseTypeHint(Stream $tokenStream)
508:     {
509:         $type = $tokenStream->getType();
510: 
511:         if (T_ARRAY === $type) {
512:             $this->typeHint = self::ARRAY_TYPE_HINT;
513:             $this->originalTypeHint = self::ARRAY_TYPE_HINT;
514:             $tokenStream->skipWhitespaces(true);
515:         } elseif (T_STRING === $type || T_NS_SEPARATOR === $type) {

517:             do {
518:                 $className .= $tokenStream->getTokenValue();

521:                 $type = $tokenStream->getType();
522:             } while (T_STRING === $type || T_NS_SEPARATOR === $type);
523: 
524:             if ('' === ltrim($className, '\\')) {
525:                 throw new Exception\ParseException($this, $tokenStream, sprintf('Invalid class name definition: "%s".', $className), Exception\ParseException::LOGICAL_ERROR);

572:      *
573:      * @param \TokenReflection\Stream\StreamBase $tokenStream Token substream
574:      * @return \TokenReflection\ReflectionParameter
575:      * @throws \TokenReflection\Exception\ParseException If the default value could not be determined.
576:      */
577:     private function parseDefaultValue(Stream $tokenStream)
578:     {
579:         if ($tokenStream->is('=')) {
580:             $tokenStream->skipWhitespaces(true);
581: 
582:             $level = 0;
583:             while (null !== ($type = $tokenStream->getType())) {
584:                 switch ($type) {
585:                     case ')':
586:                         if (0 === $level) {
587:                             break 2;
588:                         }
589:                     case '}':
590:                     case ']':
591:                         $level--;
592:                         break;
593:                     case '(':
594:                     case '{':
595:                     case '[':
596:                         $level++;
597:                         break;
598:                     case ',':
599:                         if (0 === $level) {
600:                             break 2;
601:                         }
602:                         break;
603:                     default:
604:                         break;
605:                 }

607:                 $this->defaultValueDefinition[] = $tokenStream->current();

609:             }

611:             if (')' !== $type && ',' !== $type) {
